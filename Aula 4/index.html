class ContaBancaria {
    private $saldo;

    // Construtor da classe
    public function __construct($saldoInicial) {
        $this->saldo = $saldoInicial;
    }

    // Método para depositar dinheiro na conta
    public function depositar($valor) {
        if ($valor > 0) {
            $this->saldo += $valor;
            echo "Depósito de $valor realizado com sucesso. Novo saldo: " . $this->saldo . "
";
        } else {
            echo "Erro: Valor inválido para depósito.
";
        }
    }

    // Método para sacar dinheiro da conta
    public function sacar($valor) {
        if ($valor > 0 && $valor <= $this->saldo) {
            $this->saldo -= $valor;
            echo "Saque de $valor realizado com sucesso. Novo saldo: " . $this->saldo . "
";
        } else {
            echo "Erro: Valor inválido para saque ou saldo insuficiente.
";
        }
    }

    // Método para verificar o saldo atual da conta
    public function getSaldo() {
        return $this->saldo;
    }
}

// Exemplo de uso da classe
$conta = new ContaBancaria(1000); // Cria uma conta com saldo inicial de 1000

echo "Saldo inicial: " . $conta->getSaldo() . "
"; // Exibe o saldo inicial

$conta->depositar(500); // Deposita 500 na conta
$conta->sacar(200);     // Sacar 200 da conta
$conta->sacar(2000);    // Tenta sacar um valor maior que o saldo disponível

echo "Saldo final: " . $conta->getSaldo() . "
"; // Exibe o saldo final
